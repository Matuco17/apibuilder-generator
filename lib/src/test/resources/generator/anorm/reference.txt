import anorm._

package test.apidoc.apidoctest.v0.anorm {

  package parsers {

    object Reference {

      def newParser(config: me.apidoc.lib.anorm.parsers.util.Config) = {
        config match {
          case me.apidoc.lib.anorm.parsers.util.Config.Prefix(prefix) => parser(
            guid = s"${prefix}_guid"
          )
        }
      }

      def parserByTable(table: String) = parser(
        guid = s"$table.guid"
      )

      def parser(
        guid: String
      ): RowParser[test.apidoc.apidoctest.v0.models.Reference] = {
        SqlParser.get[_root_.java.util.UUID](guid) map {
          case guid => {
            test.apidoc.apidoctest.v0.models.Reference(
              guid = guid
            )
          }
        }
      }

    }

  }

}
